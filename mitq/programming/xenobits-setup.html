create user


adduser xeno
usermod -aG root xeno
usermod -aG docker xeno
usermod -aG admin xeno
usermod -aG sudo xeno

mkdir /home/xeno/.ssh
cp .ssh/authorized_keys /home/xeno/.ssh/
sudo chown -R xeno:xeno /home/xeno/.ssh/
