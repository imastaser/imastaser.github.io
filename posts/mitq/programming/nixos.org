** enable ssh
enable sshd in hetzner, put in user data
#+BEGIN_SRC bash
# generate config file
nixos-generate-config --root /mnt
vi /mnt/etc/nixos/configuration.nix

# uncomment ssh
services.sshd.enable = true;
#+END_SRC
